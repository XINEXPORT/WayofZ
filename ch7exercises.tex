\section{Exercise 7.8.1}
Define the total operations T\_Backward and T\_Delete. You may use any function or operators defined in Appendix D.

\begin{schema}{LeftArrow}
    ch? : CHAR
    \where
    ch? = left\_arrow
\end{schema}

\begin{schema}{Backward}
    \Delta Editor \\
    ch? : CHAR
    \where
    ch? = left\_arrow \\
    right' =  \langle last(left)\rangle \cat right
    left' = front(left)
\end{schema}

\begin{zed}
    T\_Backwards \widehat{=} Backward \vee (LeftArrow \wedge \Xi Editor \wedge left = \langle\rangle)
\end{zed}

\begin{schema}{Backspace}
    ch? : CHAR
\where
    ch? = backspace\_key
\end{schema}

\begin{schema}{Delete}
    \Delta Editor \\
    Backspace
    \where
    left' = front(left) \\
    right'= right
\end{schema}

\begin{zed}
    T\_Delete \widehat{=} Delete \vee (Backspace \wedge \Xi Editor \wedge left = \langle\rangle)
\end{zed}

\section{Exercise 7.8.2}
Define the operation that handles input characters that are not handled by Insert, Forward, Backward, and Delete.

\begin{schema}{Exception}
    ch? : CHAR
    \where
    ch? \notin printing \\
    ch? \neq left\_arrow \\
    ch? \neq right\_arrow \\
    ch? \neq backspace\_key
\end{schema}

\section{Exercise 7.8.3}
Our editor would be far more useful it it could read in the contents of a file at the user's demand. Model this in Z.

%dom is the domain of a function or relation which establishes a precondition for everything on the left of dom which is f?
\begin{schema}{OpenFile}
    f? : \dom files
\end{schema}

\begin{schema}{ReadContents}
    \Delta Editor \\
    OpenFile
    \where
    left' = \langle \rangle \\
    right' = files(f?)
\end{schema}

\begin{schema}{CloseFile}
    \Delta Editor \\
    \where
    left' = \langle \rangle \\
    right' = \langle \rangle
\end{schema}
